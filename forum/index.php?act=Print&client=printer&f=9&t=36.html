Écrit par : dox 01/09/2009 - 23:37

je susi pas un spécialiste mais ca evitera de chercher partout les seta utiles

-pour debloquer les fps(indispensable)
seta com_unlock_timingMethod "1"
seta com_unlock_maxFPS "90" <-90 ou autre suivant les fps desirés
seta com_unlockFPS "1"
seta com_showfps "1"

-l'inimitable /kill ( ça servira un jour)
bind "K" "kill"

-la console par le ²
seta com_allowConsole "1"

-nom et tag
seta ui_name "tonnom" marche pas vraiment
seta ui_clanTag "R3 "
ui_clanTagPosition "0" avant le nom ou "1" après le nom

------------------------------------------------------------------------------
-les principaux bind (je colle le contenue de mon fichier bind.cfg)
unbindall
bind "z" "_forward" "" "default"
bind "s" "_back" "" "default"
bind "q" "_moveleft" "" "default"
bind "d" "_moveright" "" "default"
bind "SPACE" "_moveup" "" "default"
bind "MOUSE1" "_attack" "" "default"
bind "r" "_reload" "" "default"

/voir si je vire pour un coup de crosse
bind "MOUSE2" "_altattack_kbd" "" "default"

//voir si commande sans toggle
bind "SHIFT" "_togglesprint" "" "default"

//touches affectées à la souris
bind "M" "_throwGrenade" "" "default"
bind "L" "_use" "" "default"
bind "K" "kill"

bind "MWHEELDOWN" "_weapswap" "" "default"
bind "MWHEELUP" "_weapswap" "" "default"

bind "e" "_shroud" "" "default"
bind "3" "_shroudAbility2" "" "default"
bind "F3" "_ready" "" "default"
bind "F4" "_showScores" "" "default"
bind "TAB" "_showScores" "" "default"
bind "n" "_limbomenu" "" "default"

//chat
bind "i" "clientMessageMode" "" "default"
//teamchat
bind "o" "clientMessageMode 1" "" "default"

//pas sur de ce que ca fait alors je touche pas
bind "ALT" "_movedown_kbd" "" "default"
bind "TAB" "_menuNavForward" "" "menu"
bind "TAB" "_menuNavBackward" "SHIFT" "menu"
bind "TAB" "_menuNavBackward" "RIGHTSHIFT" "menu"
bind "ENTER" "_menuAccept" "" "menu"
bind "KP_ENTER" "_menuAccept" "" "menu"
----------------------------------------------------------------------------------------

-autoriser le alt+entree
com_useFastVidRestart "1"
